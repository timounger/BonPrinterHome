314 def read_user_file(self, s_file: str, b_remove_file: bool = False):
315 """!
316 @brief Read user file configuration and save in settings.
317 @param s_file : file top read
318 @param b_remove_file : [True] = delete file; [False] = don't delete file
319 """
320 log.debug("Import edited or new user file: %s", s_file)
321 config_data = self.handle_config_data_read(s_file)
322 if config_data is not None:
323 d_read_user_dict = config_data
324 if (not self.ui.model.c_auth.check_user_login(EUser.ADMIN)) and (EUser.ADMIN.value in d_read_user_dict):
325 del d_read_user_dict[EUser.ADMIN.value] # delete if other user write Admin PW to edit file
326 if EUser.ADMIN.value not in d_read_user_dict:
327 if EUser.ADMIN.value in self.d_user:
328 d_read_user_dict[EUser.ADMIN.value] = self.d_user[EUser.ADMIN.value] # restore admin pw if not present
329 else:
330 self.ui.set_status("No Admin data to restore", True) # state not possible
331 self.d_user = d_read_user_dict.copy()
332 # default Admin PW if not valid
333 if EUser.ADMIN.value not in self.d_user:
334 self.d_user[EUser.ADMIN.value] = D_DEFAULT_USER[EUser.ADMIN.value].copy()
335 self.d_user[EUser.ADMIN.value][S_PW] = DEFAULT_CODE
336 self.ui.set_status([f"No Admin data. Password is set to default: {DEFAULT_CODE}",
337 f"Keine Admin Daten. Passwort wurde zurückgesetzt: {DEFAULT_CODE}"], True)
338 if S_PW not in self.d_user[EUser.ADMIN.value]:
339 self.d_user[EUser.ADMIN.value][S_PW] = DEFAULT_CODE
340 self.ui.set_status([f"No Admin Password data. Password is set to default: {DEFAULT_CODE}",
341 f"Kein Admin Passwort. Passwort wurde zurückgesetzt: {DEFAULT_CODE}"], True)
342 if not self.d_user[EUser.ADMIN.value][S_PW].isdigit() and self.d_user[EUser.ADMIN.value][S_PW] != "":
343 self.d_user[EUser.ADMIN.value][S_PW] = DEFAULT_CODE
344 self.ui.set_status([f"Admin password is invalid. Password is set to default: {DEFAULT_CODE}",
345 f"Admin Passwort ist ungültig. Passwort wurde zurückgesetzt: {DEFAULT_CODE}"], True)
346 if not self.ui.test_mode:
347 self.store_user_data()
348 if b_remove_file:
349 os.remove(s_file)
350 self.ui.set_status(["User configuration changed", "Benutzer Konfiguration wurde geändert"])
351 self.check_uids()
352 else:
353 if b_remove_file:
354 b_edit = self.ui.confirm_dialog(["Invalid. Edit again?", "Ungültig. Erneut bearbeiten?"])
355 if b_edit:
356 self.ui.edit_user_in_notepad(s_file)
357 else:
358 self.ui.set_status(["Unable to read user configuration with duplicate users",
359 "Benutzerkonfiguration mit doppelten Benutzern kann nicht gelesen werden"], True)
360 else:
361 self.ui.set_status(["Unable to import user configuration with duplicate users",
362 "Importieren der Benutzerkonfiguration mit doppelten Benutzern nicht möglich"], True)
363 self.ui.unblock_ui()

459def sort_dict(d_dict_to_sort: dict, d_order_reference: dict) -> dict:
460 """!
461 @brief Sort dictionary.
462 @param d_dict_to_sort : dictionary to sort
463 @param d_order_reference : reference dictionary to sort in this order
464 @return return sorted dictionary
465 """
466 sorted_dict = OrderedDict()
467 unsorted_dict = OrderedDict()
468
469 for key, _order in d_order_reference.items():
470 if key in d_dict_to_sort:
471 value = d_dict_to_sort[key]
472
473 if isinstance(value, dict):
474 sorted_dict[key] = sort_dict(value, d_order_reference[key])
475 else:
476 sorted_dict[key] = value
477
478 # insert unsorted values at the end
479 for key, value in d_dict_to_sort.items():
480 if key not in d_order_reference:
481 if isinstance(value, dict):
482 unsorted_dict[key] = sort_dict(value, {})
483 else:
484 unsorted_dict[key] = value
485
486 sorted_dict.update(unsorted_dict)
487 return sorted_dict

490def read_data_from_config_file(s_file_name: str) -> dict:
491 """!
492 @brief Read data from configuration file.
493 @param s_file_name : file to read
494 @return return configuration of the file
495 """
496 d_data = {}
497 parser = ConfigParser()
498 with codecs.open(s_file_name, mode="r", encoding="utf-8") as file:
499 parser.read_file(file)
500 for s_section in parser.sections():
501 d_data[s_section] = {}
502 for key, value in parser.items(s_section):
503 if (key == S_ID) and value.startswith('[') and value.endswith(']'): # read UID as list if possible
504 entries = value[1:-1].split(", ")
505 entries = [entry.strip('\'\"') for entry in entries]
506 entry = entries
507 else:
508 entry = value
509 d_data[s_section][key] = entry
510 return d_data

513def write_config_to_file(s_file_name: str, d_dict: dict):
514 """!
515 @brief Create default configuration and write to file.
516 @param s_file_name : write to this file
517 @param d_dict : dictionary with default configuration to write
518 """
519 parser = cfg.ConfigParser()
520 for s_ini_section in d_dict:
521 parser.add_section(s_ini_section)
522 for s_ini_key in d_dict[s_ini_section]:
523 s_ini_value = d_dict[s_ini_section][s_ini_key]
524 if isinstance(s_ini_value, list):
525 s_ini_value = ', '.join(f'"{item}"' for item in s_ini_value)
526 s_ini_value = f"[{s_ini_value}]"
527 parser.set(s_ini_section, s_ini_key, s_ini_value)
528 with open(s_file_name, mode="w", encoding="utf-8") as o_cfg_file:
529 parser.write(o_cfg_file)
