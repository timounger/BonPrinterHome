70 def check_user_login(self, user: EUser, user_to_check: EUser = None) -> bool:
71 """!
72 @brief Check if user log in.
73 @param user : check if this user is logged in or list
74 @param user_to_check : check this user (not logged in user)
75 @return status if this user is logged in
76 """
77 b_user_loged_in = False
78 if user_to_check is None:
79 user_to_check = self.s_login_user
80 if user_to_check is not None:
81 if isinstance(user, list):
82 if user_to_check in user:
83 b_user_loged_in = True
84 elif EUser.B in user:
85 if user_to_check[0] == EUser.B:
86 b_user_loged_in = True
87 else:
88 match user:
89 case EUser.B:
90 if user_to_check[0] == user:
91 b_user_loged_in = True
92 case _:
93 if user_to_check == user:
94 b_user_loged_in = True
95 return b_user_loged_in

97 def get_user_position(self, user: EUser = None) -> int:
98 """!

105 match user:
