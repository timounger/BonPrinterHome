124 min_width: int = None, min_height: int = None, geometry: QByteArray = None, state: QByteArray = None,
125 fix_size: bool = False, resize=None, style: str = None, resize_callback: Any = None,
126 process_events: bool = False, show: bool = None, thread: bool = False):
127 """!
128 @brief Configure Window
129 @param window : window
130 @param title : windows title
131 @param icon : window icon
132 @param fullscreen : fullscreen status
133 @param min_width : minimum window width
134 @param min_height : minimum window height
135 @param geometry : window geometry
136 @param state : window state
137 @param fix_size : state if window is resizable
138 @param resize : resize size
139 @param style : window style
140 @param resize_callback : resize callback function
141 @param process_events : process event status
142 @param show : show/close window
143 @param thread : run window in Thread
144 """
145 if title is not None:
146 window.setWindowTitle(title)

148 window.setWindowIcon(QIcon(icon))
149 if fullscreen is not None:
150 if fullscreen:
151 if not window.isFullScreen():

153 else:

155 window.showNormal()
156 if min_width is not None:
157 window.setMinimumWidth(min_width)
158 if min_height is not None:

161 window.restoreGeometry(geometry)

163 window.restoreState(state)
164 if fix_size:
165 window.setFixedSize(window.size())
166 if resize is not None:
167 width, height = resize
168 window.resize(width, height)
169 if style is not None:
170 window.setStyleSheet(style)
171 if resize_callback is not None:
172 window.resized.connect(resize_callback)
173 if process_events:
174 window.processEvents()
175 if show is not None:
176 if show:
177 window.show()
178 else:
179 window.close()

343def group_available_menu(parent: MAIN_WINDOW, l_actions: List, actual_value: Any, l_match_values: List, none_action: QAction):
344 """!
345 @brief Set action group.
346 @param parent : parent window controller
347 @param l_actions: list with actions to add
348 @param actual_value : actual setting value
349 @param l_match_values : list with possible match values to check default setting
350 @param none_action : none action is set if actual_value is None
351 @return action group object
352 """
353 action_group = QActionGroup(parent)
354 action_group.addAction(none_action)
355 b_action_present = False
356 all_actions_add = True
357 config_menu(none_action, text="None")
358 if actual_value is None:
359 none_action.setChecked(True)
360 b_action_present = True
361 i_action_index = 0
362 for reader in l_match_values:
363 if i_action_index < len(l_actions):
364 config_menu(l_actions[i_action_index], enable=True, show=True, text=reader)
365 action_group.addAction(l_actions[i_action_index])
366 if actual_value == reader:
367 b_action_present = True
368 config_menu(l_actions[i_action_index])
369 l_actions[i_action_index].setChecked(True)
370 i_action_index += 1
371 else:
372 all_actions_add = False
373 break
374 for i_index in range(i_action_index, len(l_actions)):
375 if (i_index == i_action_index) and (not b_action_present):
376 config_menu(l_actions[i_index], enable=False, show=True, text=actual_value)
377 l_actions[i_index].setChecked(True)
378 else:
379 config_menu(l_actions[i_index], enable=False, show=False)
380 action_group.addAction(l_actions[i_index])
381 return action_group, all_actions_add

575def config_table(table: QTableWidget, clear_selection: bool = False, single_row_selection: bool = None, row_description: List = None, font: QFont = None,
576 size: Tuple = None, min_section_size: int = None, row_count: int = None, auto_size: bool = False):
577 """!
578 @brief Configure table
579 @param table : table to configure
580 @param clear_selection : clear selected table item
581 @param single_row_selection : only one line can select
582 @param row_description : row description text
583 @param font : row description font
584 @param size :row description size
585 @param min_section_size : default vertical size
586 @param row_count : number of rows to set
587 @param auto_size : do auto size of horizontal entries
588 """
589 if single_row_selection is not None:
590 if single_row_selection:
591 table.setSelectionMode(QAbstractItemView.SelectionMode.SingleSelection) # only one line can select
592 else:
593 table.setSelectionMode(QAbstractItemView.SelectionMode.NoSelection) # no table selection possible
594 if clear_selection:
595 table.clearSelection() # clear selected table entry to delete single items
596 if row_description is not None:
597 table.setColumnCount(len(row_description))
598 for i, value in enumerate(row_description):
599 item = QTableWidgetItem()
600 table.setHorizontalHeaderItem(i, item)
601 table.horizontalHeaderItem(i).setText(value)
602 if font is not None:
603 table.horizontalHeaderItem(i).setFont(font)
604 if size is not None:
605 x, y, width, height = size
606 table.setGeometry(QRect(x, y, width, height))
607 if min_section_size is not None:
608 # first call setMinimumSectionSize to set actual correct size
609 table.verticalHeader().setMinimumSectionSize(min_section_size) # minimum vertical size
610 table.verticalHeader().setDefaultSectionSize(min_section_size) # default vertical size
611 if row_count is not None:
612 table.setRowCount(row_count)
613 if auto_size:
614 # Update font before auto width
615 if font is not None:
616 for row in range(table.rowCount()): # set font size of table entries
617 for col in range(table.columnCount()):
618 item = table.item(row, col)
619 item.setFont(font)
620 table.resizeColumnsToContents() # auto column width
621 header = table.horizontalHeader()
622 header.setSectionResizeMode(1, QHeaderView.ResizeMode.Stretch) # fill middle row
623 table.verticalHeader().setVisible(False)
624 table.scrollToBottom() # scroll to bottom to show deposit anyway

627def insert_table_item(table: QTableWidget, i_entry_index: int, l_item_param: List, font: QFont = None, fg: str = None, bg: str = None):
628 """!
629 @brief Insert table item
630 @param table : insert item to this table
631 @param i_entry_index : table index of the new item
632 @param l_item_param : item parameter to insert to table
633 @param font : font of table item
634 @param fg : foreground color
635 @param bg : foreground color

649 item.setTextAlignment(Qt.AlignmentFlag.AlignRight | Qt.AlignmentFlag.AlignVCenter)
650 if font is not None:
651 item.setFont(font)
652 table.setItem(i_entry_index, j, item)
