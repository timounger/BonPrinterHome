77 def exception_hook(self, exc_type, exc_value, exc_traceback):
78 """!
79 @brief Custom exception hook. It is triggered each time an uncaught exception occurs.
80 @param exc_type : exception type
81 @param exc_value : exception value
82 @param exc_traceback : exception traceback
83 """
84 if issubclass(exc_type, KeyboardInterrupt):
85 # ignore keyboard interrupt to support console application
86 sys.__excepthook__(exc_type, exc_value, exc_traceback)
87 else:
88 log_msg = "\n".join([f"{exc_type.__name__}: {exc_value}",
89 "".join(traceback.format_tb(exc_traceback))])
90 log.error(log_msg)
91
92 # trigger message box show
93 self.exception_caught.emit(log_msg)
