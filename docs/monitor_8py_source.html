86 def check_for_style_change(self, e_style: ETheme = None):
87 """!
88 @brief Check for style change
89 @param e_style : style to set
90 """
91 if e_style is None:
92 e_style = self.e_style

94 match e_style:
95 case ETheme.AUTO:

97 case ETheme.LIGHT:
98 self.e_actual_theme = ETheme.LIGHT
99 case ETheme.DARK:
100 self.e_actual_theme = ETheme.DARK
101 case ETheme.CLASSIC:
102 self.e_actual_theme = ETheme.CLASSIC
103 case ETheme.SYSTEM:
104 self.e_actual_theme = ETheme.LIGHT if darkdetect.isLight() else ETheme.DARK
105 case _:
106 self.ui.set_status(f"Invalid theme change: {e_style}", True) # state not possible
107 if old_style != self.e_actual_theme:
108 self.update_icons()
109 self.set_dialog_style(self.ui)

202 def get_font(self, s_item: str, ext_factor: float = None, x_scale_factor: float = None, y_scale_factor: float = None, font: str = S_DEFAULT_FONT):
203 """!
204 @brief Get font size depend of scale factor.
205 @param s_item : item in D_FONT_SIZE with specified font size
206 @param ext_factor : extended factor (for extended button view)
207 @param x_scale_factor : x scale factor
208 @param y_scale_factor : y scale factor
209 @param font : font
210 @return return font size
211 """
212 i_size = D_FONT_SIZE[s_item]
213 x_size = i_size if (x_scale_factor is None) else i_size * x_scale_factor
214 y_size = i_size if (y_scale_factor is None) else i_size * y_scale_factor
215 i_size = min(x_size, y_size)
216 if ext_factor is not None:
217 i_size *= ext_factor
218 if s_item == "total":
219 text = get_text(self.ui.text_total)
220 lines = text.split('\n')
221 line_count = len(lines)
222 i_size /= line_count
223 i_size = int(i_size)
224 return create_font(font, i_size)

226 def resize_window(self, event=None, b_update_only_total_font: bool = False):
227 """!
228 @brief set position and size of items
229 @param event : size event with window dimension
230 @param b_update_only_total_font : status if update only text font size
231 """
232 i_actual_width, i_actual_height = get_window_size(self.ui, event)
233
234 if (i_actual_width >= I_DEFAULT_WIN_WIDTH) and (i_actual_height >= I_DEFAULT_WIN_HEIGHT):
235 b_ext_items = self.ui.model.c_config.b_extended_articles and self.ui.model.c_auth.get_article_view_state() # extended button view only in item mode with > 30
236 ext_factor = (I_BTN_BREAK / I_BTN_BREAK_EXT) if b_ext_items else 1
237 x_scale_factor = i_actual_width / I_DEFAULT_WIN_WIDTH
238 y_scale_factor = i_actual_height / I_DEFAULT_WIN_HEIGHT
239 self.table_font = self.get_font("table", ext_factor, x_scale_factor, y_scale_factor) # save as global to use when update table
240
241 if b_update_only_total_font:
242 config_text(self.ui.text_total, font=self.get_font("total", ext_factor, x_scale_factor, y_scale_factor)) # update only total font size
243 else:
244 # x position data
245 i_btn_row_cnt = I_BTN_ROW_CNT_EXT if b_ext_items else I_BTN_ROW_CNT
246 table_width = (1 / 3) * i_actual_width
247 total_button_width = i_actual_width - table_width
248 x_single_button = int(total_button_width / i_btn_row_cnt)
249
250 # y position data
251 i_btn_column_cnt = I_BTN_BREAK_EXT if b_ext_items else I_BTN_BREAK
252 total_button_high = i_actual_height - 85 # subtract size of status and menu bar
253 y_single_button = int(total_button_high / (i_btn_column_cnt + 1)) # +1 for lowest button row
254 y_order_gap = total_button_high - (y_single_button * (i_btn_column_cnt + 1))
255
256 # item buttons
257 i_row_idx = 0
258 for i, btn in enumerate(self.ui.l_bnt_fnc):
259 if i < I_NUMBER_OF_ITEMS or b_ext_items:
260 if i != 0:
261 if (i % i_btn_column_cnt) == 0:
262 i_row_idx += 1
263 i_column_idx = i % i_btn_column_cnt
264 x_pos = i_row_idx * x_single_button
265 y_pos = total_button_high - ((i_column_idx * y_single_button) + (2 * y_single_button) + y_order_gap)
266 config_btn(btn, size=(x_pos, y_pos, x_single_button, y_single_button),
267 font=self.get_font("articles_button", ext_factor, x_scale_factor, y_scale_factor))
268
269 x_table = i_actual_width - (x_single_button * i_btn_row_cnt)
270 y_table = y_single_button * i_btn_column_cnt
271 y_single_button += y_order_gap
272 y_lowest_pos = total_button_high - y_single_button
273
