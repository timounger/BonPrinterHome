218 def loop(self):
219 """!
220 @brief Printer Loop to check for items to print out.
221 """
222 s_actual_used_com_port = self.s_select_com_port
223 if (self.s_select_com_port is not None) and (self.b_select_com_port_available):
224 c_serial_port = serial.Serial(self.s_select_com_port)
225 else:
226 c_serial_port = None
227 while True:
228 if self.b_select_com_port_available and (s_actual_used_com_port != self.s_select_com_port):
229 if (c_serial_port is not None) and c_serial_port.isOpen():
230 c_serial_port.close()
231 s_actual_used_com_port = self.s_select_com_port
232 if self.s_select_com_port is not None:
233 c_serial_port = serial.Serial(self.s_select_com_port)
234 else:
235 self.s_select_com_port = None
236 if not self.bon_queue.empty() or not self.report_queue.empty() or self.b_open_drawer:
237 if c_serial_port is not None:
238 if not c_serial_port.isOpen():
239 c_serial_port.open()
240 self.check_open_drawer(c_serial_port)
241 if not self.bon_queue.empty():
242 item = self.bon_queue.get()
243 bon = self.create_bon(item)
244 for _ in range(item.count):
245 c_serial_port.write(bon)
246 time.sleep(F_PRINTER_PRINT_DELAY)
247 self.check_open_drawer(c_serial_port) # open after print same items is possible
248 elif not self.report_queue.empty():
249 report = self.report_queue.get()
250 c_serial_port.write(self.create_report_bon(report))
251 else:
252 self.ui.set_status("None printer will print. Disable printer and clear queue.", True) # state not possible
253 self.s_select_com_port = None
254 self.bon_queue = queue.Queue()
255 self.report_queue = queue.Queue()
256 self.b_open_drawer = False
257 else:
258 if (c_serial_port is not None) and c_serial_port.isOpen():
259 c_serial_port.close()
260 time.sleep(F_PRINTER_SLEEP_TIME)

279 def create_bon(self, item: Item) -> bytes:
280 """!
281 @brief Create bon code for printer.
282 @param item : item to create bon
283 @return return bon code for printer
284 """
285 c_dummy = Dummy()
286 s_pay = f"{S_UNIT} {item.price}".rjust(self.i_line_break)
287 s_info = f"{item.date} {item.user}"[:self.i_line_break]
288 s_product = self.set_umlauts(f"{item.name}"[:int(self.i_line_break / 2)]) # edit in btn_article_preview too if line break changed
289 c_dummy.set() # default values
290 if self.s_header1:
291 s_header1 = self.set_umlauts(self.s_header1.center(self.i_line_break)[:self.i_line_break])
292 c_dummy.text(f"{s_header1}\n")
293 if self.s_header2:
294 s_header2 = self.set_umlauts(self.s_header2.center(self.i_line_break)[:self.i_line_break])
295 c_dummy.text(f"{s_header2}\n")
296 c_dummy.text(f"\n{s_info}\n\n")
297 c_dummy.set(width=2, height=2)
298 c_dummy.text(f"{s_product}")
299 c_dummy.set() # default values
300 c_dummy.text(f"\n{s_pay}")
301 if self.e_printer != EPrinter.EPSON: # EPSON write empty lines at end as default
302 c_dummy.text("\n")
303 c_dummy.cut(mode="PART")
304 return c_dummy.output
