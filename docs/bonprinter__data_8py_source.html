32def resource_path(s_relative_path: str) -> str:
33 """!
34 @brief Returns the absolute path to a resource given by a relative path depending on the environment (EXE or Python)
35 @param s_relative_path : the relative path to a file or directory
36 @return absolute path to the resource
37 """
38 try:
39 # PyInstaller creates a temp folder and stores path in _MEIPASS
40 s_base_path = sys._MEIPASS # pylint: disable=no-member,protected-access
41 except BaseException:
42 s_base_path = os.path.abspath("../")
43 s_resource_path = os.path.join(s_base_path, s_relative_path)
44 log.debug("Recource Path (relative %s): %s", s_relative_path, s_resource_path)
45 return s_resource_path

48def thread_loop(obj, thread_name: str):
49 """!
50 @brief loop for threads.
51 @param obj : object of thread class
52 @param thread_name : thread name
53 """
54 try:
55 while True:
56 obj.loop()
57 except Exception:
58 exc_type, exc_obj, exc_tb = sys.exc_info()
59 fname = os.path.split(exc_tb.tb_frame.f_code.co_filename)[1]
60 s_err = f"Thread {thread_name}: {exc_type} {fname} {exc_obj} {exc_tb.tb_lineno}"
61 log.error(s_err)
62 obj.ui.qt_exception_hook.exception_caught.emit(s_err)
63 sys.exit(1)

68 @brief Show reset configuration dialog. Enter reset PW to reset configuration.

71 """

422def get_registry_value(handle: SETTINGS, s_key: str, b_none_err: bool = True) -> Any:
423 """!
424 @brief Reads the registry value with given handle and key.
425 @param handle : settings handle
426 @param s_key : get value for this key
427 @param b_none_err : [True] call error if None in setting [False] allowed None as setting
428 @return value that is mapped to the given key or raises a KeyError if key not found in handle.
429 """
430 value = handle.value(s_key, defaultValue=None)
431 if b_none_err and (value is None):
432 raise KeyError(f"{s_key} not found in group {handle.group()}")
433 return value

498def write_user_settings(d_user: dict):
499 """!
500 @brief Writes the user settings to persistent storage
501 @param d_user : user dictionary
502 """
503 dict_to_write = copy.deepcopy(d_user)
504 if platform.system() == 'Windows':
505 for user_type, user_data in dict_to_write.items(): # protect pw
506 if S_PW in user_data:
507 s_password = user_data[S_PW]
508 s_password_encrypted = win32crypt.CryptProtectData(bytes(s_password, "utf-8")) if len(s_password) > 0 else ""
509 dict_to_write[user_type][S_PW] = s_password_encrypted
510 handle = get_settings_handle()
511 handle.beginGroup(S_SECTION_CONFIGURATION)
512 handle.setValue(S_KEY_USERS, dict_to_write)
513 handle.endGroup()

516def read_user_settings() -> dict:
517 """!
518 @brief Reads the user settings from persistent storage
519 @return user settings
520 """
521 handle = get_settings_handle()
522 try:
523 handle.beginGroup(S_SECTION_CONFIGURATION)
524 d_user = get_registry_value(handle, S_KEY_USERS)
525 if platform.system() == 'Windows':
526 for user_type, user_data in d_user.items():
527 if S_PW in user_data:
528 bytes_password_enc = user_data[S_PW]
529 _, bytes_password = win32crypt.CryptUnprotectData(bytes_password_enc) if (len(bytes_password_enc) > 0) else (None, bytes(bytes_password_enc, "utf-8"))
530 d_user[user_type][S_PW] = bytes_password.decode("utf-8")
531 handle.endGroup()
532 except BaseException as e:
533 log.debug("Invalid User settings, writing and using default values (%s)", str(e))
534 d_user = D_DEFAULT_USER
535 handle.endGroup()
536 return d_user
