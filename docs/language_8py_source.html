108 language = "Unknown"

112 def get_language_text(self, l_text: list) -> str:
113 """!
114 @brief Get language text
115 @param l_text : list with possible language text
116 @return text of active language
117 """
118 index = list(ELanguages).index(self.e_language)
119 if len(l_text) > index:
120 text = l_text[index]
121 if not text:
122 text = l_text[0] # get default language if translate
123 else:
124 text = l_text[0]
125 log.debug("Not all Languages supported: %s", l_text)
126 return text
